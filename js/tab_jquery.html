タグへのアクセス

$('アクセスしたいタグをCSSセレクタの用量で指定');

タグアクセス

例

<div id="tab">

<ul class="nextBack">
	<li class="next">次へ</li>
	<li class="back">戻る</li>
</ul>

<ul class="nav">
	<li id="nav0">コンテンツ1</li>
	<li id="nav1">コンテンツ2</li>
	<li id="nav2">コンテンツ3</li>
</ul>
.
.
.
</div>
//idタブにアクセス
var tab = $('#tab');
// .navの一番最初のタグを取ってくる
var nav = $('#tab .nav');
// .nav li を取ってくる
var navBtn = $('#tab .nav li');

//「,」くぎりで複数指定可能
var tabBack = $('#tab .next,#tab .back');

クリック時に関数を実行

$('タグ').click(function () {
処理
});

例

$('#tab .nav li').click(function () {
	console.log('test');
});
