for..in文と連想配列

for..in文

for(変数名 in 連想配列（Object）) {
	処理
}

例

var fruitsObj = {
	red : ['りんご','さくらんぼ','いちご'],
	orange : ['かき','みかん','びわ'],
	yellow : ['バナナ','パイナップル','レモン']
};


for(var fruit in fruitsObj ) {
	console.log(fruit);
	console.log(fruitsObj[fruit])
}

表示結果
red
["りんご", "さくらんぼ", "いちご"]
orange
["かき", "みかん", "びわ"]
yellow
["バナナ", "パイナップル", "レモン"]
